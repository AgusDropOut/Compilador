MAIN {
    ulong CONTADOR,X,Y;
    CONTADOR := -5.4D-4 + 4UL;
    X,Y = 1UL, 5UL;
    while (CONTADOR < 10UL) do {
        if (X < 5UL) {
            while (Y < 3UL) do {
                if (X > Y) {
                    X := X + 1UL;
                } else {
                    Y := Y + 1UL;
                } endif;
            };
        } else {
            CONTADOR := CONTADOR + 1UL;
        } endif;
    };

}
